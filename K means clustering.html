cs = []
for i in range(1, 11):
    kmeans = KMeans(n_clusters = i, init = 'k-means++', max_iter = 300, n_init = 10, random_state = 0)
    kmeans.fit(X,y)
    cs.append(kmeans.inertia_)
plot.plot(range(1, 11), cs)
plot.title('The Elbow Method')
plot.xlabel('Value of K')
plot.ylabel('CS')
plot.show()
